clear variables; close all; clc

% simulate a continuous signal
A1=5; f1=3;
w1=2*pi*f1;

dt_sim=0.001; t_stop=6;
t_sim=0:dt_sim:t_stop;
y_sim=A1*sin(w1*t_sim);
\end{lstlisting}

plot(t_sim,y_sim,'-',t_sam,y_sam,'o')
