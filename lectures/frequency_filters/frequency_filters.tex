\begin{multicols}{2}

First Order Model
\[ \tau \dot{y}+y=KA \sin(\omega t) \]

Response Equation
\includegraphics[scale=.5]{images/rc_circuit.png} \vspace{5mm} \\
\end{multicols}

\[ y(t)=Ce^{-\frac{t}{\tau}}+\frac{KA}{\sqrt{1+(\omega\tau)^2}}\sin(\omega t-\tan^{-1}(\omega\tau)) \]
